### Basic Wrestler Creation

Simple wrestler creation for quick roster building:

```rust
#[component]
pub fn CreateWrestlerForm() -> impl IntoView {
    let (name, set_name) = create_signal(String::new());
    let (gender, set_gender) = create_signal("Male".to_string());
    let (creating, set_creating) = create_signal(false);
    
    let create_wrestler_action = move |_| {
        if name.get().trim().is_empty() { return; }
        
        set_creating.set(true);
        let wrestler_data = WrestlerData {
            name: name.get().trim().to_string(),
            gender: gender.get(),
        };
        
        spawn_local(async move {
            match create_wrestler(wrestler_data).await {
                Ok(new_wrestler) => {
                    log::info!("Wrestler created: {}", new_wrestler.name);
                    set_name.set(String::new());
                }
                Err(e) => log::error!("Failed to create wrestler: {}", e),
            }
            set_creating.set(false);

    view! {
        <div class="bg-slate-800 rounded-lg p-6">
            <h3 class="text-xl font-bold text-yellow-400 mb-4">"Create New Wrestler"</h3>
            
            <div class="space-y-4">
                <div>
                    <label class="label">
                        <span class="label-text">"Wrestler Name"</span>
                    </label>
                    <input
                        type="text"
                        placeholder="Enter wrestler name..."
                        prop:value=name
                        on:input=move |ev| set_name.set(event_target_value(&ev))
                        class="input input-bordered w-full"
                    />
                </div>
                
                <div>
                    <label class="label">
                        <span class="label-text">"Gender"</span>
```

The wrestler details window provides complete wrestler management:

```rust
#[component]
pub fn WrestlerDetailsWindow() -> impl IntoView {
    let (wrestler, set_wrestler) = create_signal(None::<Wrestler>);
    let (loading, set_loading) = create_signal(true);
    let (error, set_error) = create_signal(None::<String>);
    
    // Extract wrestler ID from URL
    let wrestler_id = use_memo(move |_| extract_wrestler_id_from_url());
    
    // Load wrestler data
    Effect::new(move |_| {
        if let Some(id) = wrestler_id.get() {
            set_loading.set(true);
            spawn_local(async move {
                match get_wrestler_by_id(id).await {
                    Ok(Some(w)) => {
                        set_wrestler.set(Some(w));
                        set_error.set(None);
                    }
                    Ok(None) => set_error.set(Some("Wrestler not found".to_string())),
                    Err(e) => set_error.set(Some(e)),
                }
                set_loading.set(false);
            });
        }
    });
    
    view! {
        <div class="min-h-screen bg-slate-900 text-white p-6">
            <Show when=move || !loading.get()>
                <Show
                    when=move || error.get().is_none()
                    fallback=move || view! {
                        <div class="text-center mt-20">
                            <div class="text-red-400 text-xl mb-4">
                                {error.get().unwrap_or_default()}
                            </div>
                            <button 
                                on:click=move |_| {
                                    if let Some(id) = wrestler_id.get() {
                                        spawn_local(async move {
                                            set_loading.set(true);
                                            // Retry loading
                                        });
                                    }
                                }
                                class="btn btn-primary"
                            >
                                "Retry"
                            </button>
                        </div>
                    }
                >
                    <div class="max-w-4xl mx-auto space-y-6">
                        // Header Section
                        <WrestlerHeader wrestler=wrestler />
                        
                        // Main Content Grid
                        <div class="grid grid-cols-1 lg:grid-cols-2 gap-6">
                            // Left Column
                            <div class="space-y-6">
                                <BasicInfoSection wrestler=wrestler set_wrestler=set_wrestler />
                                <PowerRatingsSection wrestler=wrestler set_wrestler=set_wrestler />

                                <SignatureMovesSection wrestler=wrestler />

                        // Action Buttons

            </Show>
            
            <Show when=loading>
                <div class="flex justify-center items-center min-h-screen">
                    <span class="loading loading-spinner loading-lg"></span>
                </div>
            </Show>
        </div>
    }
}
```

### Wrestler List Interface

Mass wrestler management with filtering and sorting:

```rust
#[component]
pub fn WrestlerListManagement() -> impl IntoView {
    let (wrestlers, set_wrestlers) = create_signal(Vec::<Wrestler>::new());
    let (search_filter, set_search_filter) = create_signal(String::new());
    let (gender_filter, set_gender_filter) = create_signal("All".to_string());
    let (sort_by, set_sort_by) = create_signal("name".to_string());
    let (loading, set_loading) = create_signal(true);
    
    // Load all wrestlers
    Effect::new(move |_| {
        set_loading.set(true);
        spawn_local(async move {
            match get_wrestlers().await {
                Ok(wrestler_list) => set_wrestlers.set(wrestler_list),
                Err(e) => log::error!("Failed to load wrestlers: {}", e),
            }
            set_loading.set(false);
        });
    });
    
    let filtered_wrestlers = move || {
        let mut wrestlers_list = wrestlers.get();
        let search = search_filter.get().to_lowercase();
        let gender = gender_filter.get();
        
        // Apply filters
        wrestlers_list.retain(|w| {
            let name_match = w.name.to_lowercase().contains(&search);
            let gender_match = gender == "All" || w.gender == gender;
            name_match && gender_match
        });
        
        // Apply sorting
        match sort_by.get().as_str() {
            "name" => wrestlers_list.sort_by(|a, b| a.name.cmp(&b.name)),
            "rating" => wrestlers_list.sort_by(|a, b| calculate_overall_rating(b).cmp(&calculate_overall_rating(a))),
            "wins" => wrestlers_list.sort_by(|a, b| b.wins.cmp(&a.wins)),
            "winrate" => wrestlers_list.sort_by(|a, b| {
                let a_rate = if a.wins + a.losses > 0 { a.wins as f32 / (a.wins + a.losses) as f32 } else { 0.0 };
                let b_rate = if b.wins + b.losses > 0 { b.wins as f32 / (b.wins + b.losses) as f32 } else { 0.0 };
                b_rate.partial_cmp(&a_rate).unwrap_or(std::cmp::Ordering::Equal)
            }),
            _ => {}
        }
        
        wrestlers_list
    };
    
    view! {
        <div class="space-y-6">
            // Filters and Controls
            <div class="bg-slate-800 rounded-lg p-4">
                <div class="grid grid-cols-1 md:grid-cols-4 gap-4">
                    <input
                        type="text"
                        placeholder="Search wrestlers..."
                        prop:value=search_filter
                        on:input=move |ev| set_search_filter.set(event_target_value(&ev))
                        class="input input-bordered"
                    />
                    <select
                        prop:value=gender_filter
                        on:change=move |ev| set_gender_filter.set(event_target_value(&ev))
                        class="select select-bordered"
                    >
                        <option value="All">"All Genders"</option>
                        <option value="Male">"Male"</option>
                        <option value="Female">"Female"</option>
                    </select>
                    <select
                        prop:value=sort_by
                        on:change=move |ev| set_sort_by.set(event_target_value(&ev))
                        class="select select-bordered"
                    >
                        <option value="name">"Sort by Name"</option>
                        <option value="rating">"Sort by Rating"</option>
                        <option value="wins">"Sort by Wins"</option>
                        <option value="winrate">"Sort by Win Rate"</option>
                    </select>
                    <div class="text-sm text-slate-400 flex items-center">
                        {move || filtered_wrestlers().len()} " wrestlers"
                    </div>
                </div>
            </div>
            
            // Wrestler Grid
            <Show when=move || !loading.get()>
                <div class="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-4">
                    <For
                        each=filtered_wrestlers
                        key=|wrestler| wrestler.id
                        children=move |wrestler| view! {
                            <WrestlerCard wrestler=wrestler show_actions=true />
                        }
                    />
                </div>
            </Show>
            
            <Show when=loading>
                <div class="flex justify-center py-20">
                    <span class="loading loading-spinner loading-lg"></span>
                </div>
            </Show>
        </div>
    }
}
```
